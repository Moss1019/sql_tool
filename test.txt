
user {
  user_id int primary auto_increment
  first_name string
  last_name string
  user_name string unique
  email string
}

item {
  item_id int primary auto_increment
  user_id int foreign
}

item_collaborator {
  user_id int foreign(user_id)
  item_id int foreign(item_id)
  joined user_item
}

collaborator {
  user_id int
  collaborator_id int foreign(user_id)
}
