

user {
  user_id int primary auto_increment
  first_name string
  last_name string
  user_name string unique
  email string
}

credential {
  credential_id int primary auto_increment
  password_hash string
  user_id int foreign
}

item {
  item_id int primary auto_increment
  title string
  description string
  user_id int foreign
  date_created date
}

milestone {
  milestone_id int primary auto_increment
  description string
  is_completed boolean
  item_id int foreign
}
