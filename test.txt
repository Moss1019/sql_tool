
user {
  user_id int primary auto_increment
  first_name string
  last_name string
  user_name string unique
  email string
}

item {
  item_id int primary auto_increment
  title string
  user_id int foreign
}

milestone {
  milestone_id int primary auto_increment
  title string
  date_created date
  is_complete boolean
  item_id int foreign
}

item_collaborator {
  user_id int foreign(user_id)
  item_id int foreign(item_id)
  joined user_item
}

collaborator {
  user_id int foreign(user_id)
  collaborator_id int foreign(user_id)
}
